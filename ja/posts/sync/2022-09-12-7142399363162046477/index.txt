<h3 id="二性能优化">二、性能优化</h3>
<h4 id="1公共组件性能优化">1、公共组件性能优化</h4>
<pre><code>&lt;template&gt;
    &lt;div :id=&quot;id&quot; style=&quot;display: inline-block;vertical-align: middle;&quot;&gt;&lt;/div&gt;
&lt;/template&gt;

&lt;script&gt;
import lottie from &quot;lottie-web&quot;;
export default {
    name: &quot;MyLottie&quot;,
    title: &quot;这里是页面或组件名字&quot;,
    data() {
        return {
            id:`lottie_${Math.uuidFast(&quot;&quot;)}`
        };
    },
    props:{
        animationJSON:[String, Object]
    },
    mounted() {
        this.loadAnimation();
    },
    created() {
    },
    methods: {
        loadAnimation(){
            let { animationJSON } = this;
            let lottieBox = null;

            // loop: 是否循环 autoplay: 是否自动播放
            let animationParams = { loop: false, autoplay: true };

            /**
             * 根据传进来的类型，设置不同的属性，
             * path 传 JSON文件路径，animationData 传JSON数据，与path互斥
             */
            if(Object.prototype.toString.call(animationJSON) === &quot;[object String]&quot;){
                animationParams.path = animationJSON;
            }else{
                animationParams.animationData = animationJSON;
            }

            console.log(&quot;加载动画需要的参数&quot;, animationParams);
            lottieBox = lottie.loadAnimation({
                container: document.getElementById(this.id), 
                renderer: &quot;svg&quot;, // 渲染方式svg
                ...Object.assign(animationParams, this.$attrs)
            });

            // DOMLoaded DOM元素加载完成时触发
            lottieBox[&quot;addEventListener&quot;](&quot;DOMLoaded&quot;, () =&gt; {
                let animationData = JSON.stringify(lottieBox.animationData); // json动画资源
                if(Object.prototype.toString.call(animationJSON) === &quot;[object String]&quot;){
                    localStorage.setItem(lottieBox.fileName, animationData);
                }
            });
        }
    }
};
&lt;/script&gt;
