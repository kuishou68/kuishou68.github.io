import re
import json
import time
import requests

# 循环遍历分页内容
for page in range(1,10):
    try:
        # 目标
        url=f"https://www.guo68.com/market?page={page}"
        # action
        response=requests.get(url)

        print(response.text)
        # 获取标签里的内容
        a=re.findall('<li class="list1">(.*?)</li>',response.text)[1:]
        b=re.findall('<li class="list2 ellipsis">(.*?)</li>',response.text)
        c=re.findall('<li class="list3">(￥\d.*/斤)</li>',response.text)[1:]
        d=re.findall('<li class="list6">(.*?)</li>',response.text)[1:]

        print(a,b,c,d)
        # 对数据进行筛选，写入json文件
        for name,city,price,date in zip(a,b,c,d):
            print(name,city,price,date)
            if name=="其他":
                continue
            data={"name":name,"city":city,"price":price,"date":date}
            with open("fruit.json",'a',encoding="utf-8")as f:
                json.dump(data,f,ensure_ascii=False,indent=1)
        # 每次执行完等待时间
        time.sleep(3)
    except:
        continue

            month() {
                this.days = '月';
                let monthData = [];
                let monthOrderData = []; // 每月订单数
                let monthPayData = []; // 每月支付数
                let monthCancelData = []; // 每月取消数
                // 按月统计订单数、支付数、取消数
                this.orderList.forEach((item) => {
                    if (item.order_cnt) {
                        monthOrderData.push({
                            month: item.time.substring(0, 7),
                            order_cnt: item.order_cnt,
                        });
                    }
                    if (item.pay_cnt) {
                        monthPayData.push({
                            month: item.time.substring(0, 7),
                            pay_cnt: item.pay_cnt,
                        });
                    }
                    if (item.cancel_cnt) {
                        monthCancelData.push({
                            month: item.time.substring(0, 7),
                            cancel_cnt: item.cancel_cnt,
                        });
                    }
                });
                // 统计每个月的订单数
                monthOrderData = monthOrderData.reduce((obj, item) => {
                    let find = obj.find(i => i.month === item.month);
                    let d = {
                        ...item,
                        frequency: 1
                    };
                    find ? (find.order_cnt += item.order_cnt, find.frequency++) : obj.push(d);
                    return obj
                }, []);
                // 统计每个月的支付数
                monthPayData = monthPayData.reduce((obj, item) => {
                    let find = obj.find(i => i.month === item.month);
                    let d = {
                        ...item,
                        frequency: 1
                    };
                    find ? (find.pay_cnt += item.pay_cnt, find.frequency++) : obj.push(d);
                    return obj
                }, []);
                // 统计每个月的取消数
                monthCancelData = monthCancelData.reduce((obj, item) => {
                    let find = obj.find(i => i.month === item.month);
                    let d = {
                        ...item,
                        frequency: 1
                    };
                    find ? (find.cancel_cnt += item.cancel_cnt, find.frequency++) : obj.push(d);
                    return obj
                }, []);
                // 合并同月的订单数、字符数、取消数
                monthOrderData.forEach((item) => {
                    monthPayData.forEach((i) => {
                        if (i.month === item.month) {

                    let find = obj.find(i => i.year === item.year);
                    let d = {
                        ...item,
                        frequency: 1
                    };
                    find ? (find.order_cnt += item.order_cnt, find.frequency++) : obj.push(d);
                    return obj
                }, []);
