import moment from &quot;moment&quot;;
export default {
  name: &quot;DatePicker&quot;,
  model: {
    prop: &quot;searchDate&quot;,
    event: &quot;selectDate&quot;
  },
  props: {
    searchDate: {
      type: String,
      required: true
    }
  },
  data() {
    return {
      visible: true,
      weeks: [&quot;一&quot;, &quot;二&quot;, &quot;三&quot;, &quot;四&quot;, &quot;五&quot;, &quot;六&quot;, &quot;日&quot;],
      days: [],
      currentDay: 1,
      currentMonth: 1,
      currentYear: 2022,
      currentWeek: 1,
      currYearMonthDay: undefined,
      selectDay: undefined,
      newDate: moment(new Date()).format(&quot;YYYY-MM-DD&quot;)
    };
  },
  created() {},
  mounted() {
    this.currYearMonthDay = this.searchDate; // 当前年月日
    let dateArr = this.searchDate.split(&quot;-&quot;);
    this.selectDay = dateArr[dateArr.length - 1]; // 当月的几号
    this.initData(null);
  },
  methods: {
    // 初始化日历（按月加载，月初1号前置灰部分根据当天是周几开始往前补，月末从下月第一天补）
    initData(cur) {
      let date;
      if (cur) {
        date = new Date(cur.replace(/-/g, &quot;/&quot;));
      } else {
        let now = new Date();
        let d = new Date(
          this.formatDate(now.getFullYear(), now.getMonth(), 1).replace(
            /-/g,
            &quot;/&quot;
          )
        );
        d.setDate(41); // 每次加载6周，每周7天
        date = new Date(
          this.formatDate(d.getFullYear(), d.getMonth() + 1, 1).replace(
            /-/g,
            &quot;/&quot;
          )
        );
      }
      this.currentYear = date.getFullYear();
      this.currentMonth = date.getMonth() + 1;
      this.currentDay = date.getDate();
      this.currentWeek = date.getDay() - 1;
      // 如果本月1号为周日，往前补6天
      if (this.currentWeek === -1) {
        this.currentWeek = 6;
      }
      let str = this.formatDate(
        this.currentYear,
        this.currentMonth,
        this.currentDay
      );
      this.days.length = 0;
      let daysArr = [];
      // 先获取本月1号为周几，再往前补满本周
      for (let i = this.currentWeek; i &gt;= 0; i--) {
        let d2 = new Date(str.replace(/-/g, &quot;/&quot;));
        d2.setDate(d2.getDate() - i);
        let dayObjectSelf = {}; // 用一个对象包装Date对象，便为以后预定功能添加属性
        dayObjectSelf.day = d2;
        daysArr.push(dayObjectSelf); // 将日期放入data 中的days数组 供页面渲染使用
      }
      // 其他周
      for (let j = 1; j &lt;= 41 - this.currentWeek; j++) {
        let d3 = new Date(str.replace(/-/g, &quot;/&quot;));
        d3.setDate(d3.getDate() + j);
        let dayObjectOther = {};
        dayObjectOther.day = d3;
        daysArr.push(dayObjectOther);
      }
      this.days = daysArr;
    },
    // 返回 类似 2022-10-15 格式的字符串
    formatDate(year, month, day) {
      let y = year;
      let m = month;
      if (m &lt; 10) {
        m = &quot;0&quot; + m;
      }
      let d = day;
      if (d &lt; 10) {
        d = &quot;0&quot; + d;
      }
      return y + &quot;-&quot; + m + &quot;-&quot; + d;
    },
    getDayTime(el, index) {
      let value = moment(el).format(&quot;YYYY-MM-DD&quot;);
      this.$emit(&quot;selectDate&quot;, value);
      this.cancel();

    // 上一天
    pickPreDay(year, month, day) {
      if (year &amp;&amp; month &amp;&amp; day) {
        let value = &quot;&quot;;
        let daysArr = this.days; // 日历数组（仅展示当月）
        let currDay = this.formatDate(year, month, Number(day)); // 当前日期
        // 日历中加载到第一天时，初始化日历更新 days
        if (moment(daysArr[0].day).format(&quot;YYYY-MM-DD&quot;) === currDay) {
          // 日历第一天是周一并且为当月1号，则更新上月日历
          if (daysArr[0].day.getDay() === 1 &amp;&amp; daysArr[0].day.getDate() === 1) {
            month--;
          }
          daysArr = this.initData(this.formatDate(year, month, 1));
        }
        // console.log(daysArr);
        if (daysArr?.length &gt; 0) {
          for (let i = 1; i &lt; daysArr.length; i++) {
            if (moment(daysArr[i].day).format(&quot;YYYY-MM-DD&quot;) === currDay) {
              this.currentYear = daysArr[i - 1].day.getFullYear();
              this.currentMonth = daysArr[i - 1].day.getMonth() + 1;
              this.selectDay = daysArr[i - 1].day.getDate();
              value = this.formatDate(
                this.currentYear,
                this.currentMonth,
                this.selectDay
              );
              // console.log(value);
              this.$emit(&quot;selectDate&quot;, value);
            }
          }
        }
      }
    },
    // 下一天
    pickNextDay(year, month, day) {
      if (year &amp;&amp; month &amp;&amp; day) {
        let value = &quot;&quot;;
        let daysArr = this.days; // 日历数组（仅展示当月）
        let currDay = this.formatDate(year, month, Number(day)); // 当前日期
        // 日历中加载到倒数第二天时，初始化日历更新 days
        if (
          moment(daysArr[daysArr.length - 2].day).format(&quot;YYYY-MM-DD&quot;) ===
          currDay
        ) {
          daysArr = this.initData(this.formatDate(year, month, 1));
        }
        // console.log(daysArr);
        if (daysArr?.length &gt; 0) {
          for (let i = 1; i &lt; daysArr.length; i++) {
            if (moment(daysArr[i].day).format(&quot;YYYY-MM-DD&quot;) === currDay) {
              this.currentYear = daysArr[i + 1].day.getFullYear();
              this.currentMonth = daysArr[i + 1].day.getMonth() + 1;
              this.selectDay = daysArr[i + 1].day.getDate();
              value = this.formatDate(
                this.currentYear,
                this.currentMonth,
                this.selectDay
              );
              // console.log(value);
              this.$emit(&quot;selectDate&quot;, value);
            }
          }
        }
      }
    },
    // 上个月、去年
    pickPre(year, month) {
      // setDate(0); 上月最后一天 setDate(-1); 上月倒数第二天 setDate(dx) 参数dx为 上月最后一天的前后dx天
      if (year &amp;&amp; month) {
        let d = new Date(this.formatDate(year, month, 1).replace(/-/g, &quot;/&quot;));
        d.setDate(0);
        this.currYearMonthDay = this.formatDate(
          d.getFullYear(),
          d.getMonth() + 1,
          1
        );
        this.initData(this.formatDate(d.getFullYear(), d.getMonth() + 1, 1));
      }
    },
    // 下个月、明年
    pickNext(year, month) {
      if (year &amp;&amp; month) {
        let d = new Date(this.formatDate(year, month, 1).replace(/-/g, &quot;/&quot;));
        d.setDate(41);
        this.currYearMonthDay = this.formatDate(
          d.getFullYear(),
          d.getMonth() + 1,
          1
        );
        this.initData(this.formatDate(d.getFullYear(), d.getMonth() + 1, 1));
      }
    },
    // 是否为查询日期
    isSelectDate(day) {
      let { searchDate } = this;
      let ergodicDate = this.formatDate(
        day.getFullYear(),
        day.getMonth() + 1,
        day.getDate()
      );
      return searchDate == ergodicDate;
    },
    // 是否为当天
    isActive(day) {
      return (
        day.getFullYear() == new Date().getFullYear() &amp;&amp;
        day.getMonth() == new Date().getMonth() &amp;&amp;
        day.getDate() == new Date().getDate()
      );
    },
    handlePickerDialog(searchDate) {
      let strArr = searchDate.split(&quot;-&quot;);
      this.visible = true;
      this.currYearMonthDay = searchDate;
      this.initData(this.formatDate(strArr[0], strArr[1], 1));
    },
    cancel() {
      // this.visible = false;
    }
  }
};
