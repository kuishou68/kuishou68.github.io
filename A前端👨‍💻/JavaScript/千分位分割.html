function splitByDot(num = '') {
    let temp = num.split('.');
    let toDo;
    if (temp.length === 2) {
        toDo = temp[0]
    } else {
        toDo = temp;
    }
    toDo = toDo.toString();
    let s = '';
    for (let index = toDo.length -1; index >= 0; index--) {
        s = toDo.charAt(index) + s;
        if (index % 3 === 0 && index !== 0) {
            s = ',' + s
        }
    }
    return temp[1] ? s + '.' +temp[1].toString() :s;
}
