function isFunction(value) {

function MyPromise(executor) {

        setTimeout(() => {
            if(this.status === PENDING) {
                this.status = FULFILLED;
                this.value = value;
                this.onfulfilledList.forEach(cb => {
                    cb(this.value)
                })
            }
        }, 0)

                this.status = REJECTED;
                this.reason = value;
                this.onrejectedList.forEach(cb => {
                    cb(this.reason)
                })
            }

        executor(resolve, reject)

    }
}
