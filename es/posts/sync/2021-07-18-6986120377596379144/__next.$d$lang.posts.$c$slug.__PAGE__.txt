<p><strong>实现思路</strong>：用到了《数据结构》中<strong>栈</strong>的原理，遇到
<code>#</code> 号进栈，遇到 <code>/</code> 线出栈； <code>#</code>
号标记的tokens，需要递归处理它的下标为2的小数组。遍历传入的 tokens
的每一个 token，遇到第一项是 # 和 /
的分别做处理，其余的做一个默认处理。大致思路是当遍历到的 token
的第一项为 # 时，就把直至遇到配套的 / 之前，遍历到的每一个 token
都放入一个容器（collector）中，把这个容器放入当前 token 里作为第 3
项元素。</p>
<pre class="arduino"><code>// nestTokens.js
export default (tokens) =&gt; {
  const nestTokens = []
  const stack = []
  let collector = nestTokens // 一开始让收集器 collector 指向最终返回的数组 nestTokens
  tokens.forEach(token =&gt; {
    switch (token[0]) {
      case &#39;#&#39;:
        stack.push(token)
        collector.push(token)
        collector = token[2] = [] // 连等赋值
        break
      case &#39;/&#39;:
        stack.pop(token)
        collector = stack.length &gt; 0 ? stack[stack.length-1][2] : nestTokens
        break;
      default:
        collector.push(token)
        break
    }
  })
  return nestTokens
}</code></pre>
