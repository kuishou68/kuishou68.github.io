<h4 id="问题描述">问题描述:</h4>
<p>第一次打开分页器的时候能正常看到分页器，第二次打开分页器神奇般的不见了，最开始我判断的方向有两个：</p>
<p>①分页器初始化的时候，swiper 在 imgList 还没获取到就提前加载了；</p>
<p>②关闭分页器的时候，swiper
未能完全销毁，重新切换一个tab页或者刷新网页又能出现；</p>
<p>在查阅了很多资料之后，发现以上两个判断都咩有准确的解决方案，干脆一不做二不休，自定义一个分页器；</p>
<h4 id="解决方案自定义分页器">解决方案：自定义分页器</h4>
<pre><code>&lt;!-- mySwiper 自定义弹窗 --&gt;

    &lt;a-modal class=&quot;payedOrderModal&quot; :keyboard=&quot;false&quot; :maskClosable=&quot;true&quot; :footer=&quot;null&quot; :destroyOnClose=&quot;!0&quot; centered :body-style=&quot;{paddingTop:0,paddingBottom:0,height:&#39;500px&#39;,minHeight:&#39;500px&#39;}&quot; :width=&quot;990&quot; :visible=&quot;showFlag&quot; cancelText=&quot;取消&quot; @cancel=&quot;cancel&quot;&gt;
        &lt;div slot=&quot;title&quot;&gt;图片&lt;/div&gt;

                &lt;div class=&quot;img swiper-zoom-container&quot;&gt;
                    &lt;img :src=&quot;item.url || item&quot; alt=&quot;&quot;&gt;
                &lt;/div&gt;

import &quot;swiper/css/swiper.css&quot;;
import { Swiper, SwiperSlide } from &quot;vue-awesome-swiper&quot;;
export default {
    name: &quot;MySwiper&quot;,
    title: &quot;这里是页面或组件名字&quot;,
    components: {
        Swiper,
        SwiperSlide
    },
    props:{
                // 当前图片索引
        activeIndex:{
            type:Number,
            default(){
                return 0;
            }
        },
                // 所有图片列表
        imgList:{
            type:Array,
            default(){
                return [];
            }
        }
    },
    data() {
        return {
            showFlag: true,
            currentIndex: null,
            bulletActive: &quot;swiper-pagination-bullet-active&quot;,
            swiperOptions: {
                initialSlide:0,
                loop: true,
                autoplay: false,
                zoom: {
                    minRatio:0.8,
                    maxRatio:3
                },
                navigation: {
                    nextEl: &quot;.swiper-button-next&quot;,
                    prevEl: &quot;.swiper-button-prev&quot;
                },
                observer: true,
                observeParents: true
            }
        };
    },
    created() {
    },
    mounted() {
        this.swiperOptions.initialSlide = this.activeIndex;
        this.currentIndex = this.activeIndex;
    },
    methods: {
        onSwiper(swiper) {
            console.log(swiper);
        },
        onSlideChange() {
            console.log(&quot;slide change&quot;);
        },
        prev(){
            let { realIndex } = this.$refs.mySwiper.$swiper;
            realIndex = !realIndex ? this.imgList?.length : this.currentIndex;
            this.$set(this, &quot;currentIndex&quot;, realIndex - 1);
            this.$refs.mySwiper.$swiper.slidePrev();
        },
        next(){
            let { realIndex } = this.$refs.mySwiper.$swiper;
            realIndex = realIndex === this.imgList?.length - 1 ? -1 : this.currentIndex;
            this.$set(this, &quot;currentIndex&quot;, realIndex + 1);
            this.$refs.mySwiper.$swiper.slideNext();
        },
        selectPagination(index){
            this.currentIndex = index;
            this.$refs.mySwiper.$swiper.slideTo(index + 1);
        },
        cancel(){
            this.showFlag = false;
            this.$emit(&quot;cancel&quot;);
        }
    },
    beforeDestroy() {
        this.$refs.mySwiper.$swiper.destroy();
    }
};
