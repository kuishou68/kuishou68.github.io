export default function vitePluginLvdeploy(enforce) {

import fs from &quot;fs&quot;
import archiver from &quot;archiver&quot;;
// 初始化SSH连接

function inquiry() {
  console.log(`${__dirname}/${paperFileName}.zip`)
  // 创建文件输出流
  let output = fs.createWriteStream(`${__dirname}/${paperFileName}.zip`);
  let archive = archiver(&#39;zip&#39;, {

  // 文件输出流结束

function detectionDeploy() {
  if(Object.keys(DeployConfig).length==0){
    console.log(&quot;\x1B[33m%s\x1b[0m&quot;, &#39;已压缩本地zip包服务器未配置无法部署&#39;);

  readline.question(`?请输入希望的部署环境  ${Object.keys(DeployConfig).join(&#39; 或 &#39;)} (0表示不希望压缩zip和部署) \n`, environmenttxt =&gt; {
    let environment = environmenttxt.trim();//处理首位空格
    if (environment in DeployConfig) {
      if (
        DeployConfig[environment].host &amp;&amp;
        DeployConfig[environment].port &amp;&amp;
        DeployConfig[environment].username &amp;&amp;
        DeployConfig[environment].password &amp;&amp;
        DeployConfig[environment].serverpath
      ) {
        // 参数有效
        connect(DeployConfig[environment]);
      } else {
        console.log(&quot;\x1B[33m%s\x1b[0m&quot;, &#39;Error:请检查传入参数是否规范&#39;)
      }
    } else if (environment == 0) {
      console.log(&quot;\x1B[33m%s\x1b[0m&quot;, &#39;Error:部署中断&#39;)
    } else {
      console.log(&quot;\x1B[33m%s\x1b[0m&quot;, &#39;Error:意料之外的环境&#39;)

function connect(serverConfig) {
  console.log(&quot;\x1B[36m%s\x1b[0m&quot;, &quot;-----------准备连接服务器-----------&quot;);
  conn.on(&#39;ready&#39;, () =&gt; {
    console.log(&quot;\x1B[36m%s\x1b[0m&quot;, &quot;-----------连接服务器成功 准备上传文件-----------&quot;);
    // 保证服务器下当前目录存在
    useExec(`mkdir -p ${serverConfig.serverpath}/ &amp;&amp; cd ${serverConfig.serverpath}/ &amp;&amp; rm -rf *`).then(() =&gt; {
      conn.sftp((err, sftp) =&gt; {
        sftp.fastPut(`${__dirname}/${paperFileName}.zip`, `${serverConfig.serverpath}/${paperFileName}.zip`, {}, (err, result) =&gt; {
          console.log(&quot;\x1B[32m%s\x1b[0m&quot;, `文件上传完成 : ${__dirname}/${paperFileName}.zip 已部署服务器 ${serverConfig.serverpath}/${paperFileName}.zip`);
          // 解压文件停止连接数据库
          useExec(`
              cd ${serverConfig.serverpath}/ &amp;&amp; \
              unzip ${paperFileName}.zip &amp;&amp; \
              rm -rf ${paperFileName}.zip &amp;&amp; \
              exit
            `).then(() =&gt; {
            conn.end();
            console.log(&quot;\x1B[35m%s\x1b[0m&quot;, `部署完成喽`);
          })

        })
      })
    }).catch(err =&gt; { })
  }).on(&quot;error&quot;, err =&gt; {
    console.log(&quot;\x1B[31m%s\x1b[0m&quot;, `连接服务器失败${err.toString()}`);
  }).connect({
    host: serverConfig.host,
    port: serverConfig.port,
    username: serverConfig.username,
    password: serverConfig.password
  })
}


function useExec(cmd) {
  // 所有错误命令直接关闭服务器连接
  return new Promise((resolve, reject) =&gt; {
    conn.exec(cmd, async (err, stream) =&gt; {
      if (err) { // 异常抛出
        console.log(&quot;\x1B[35m%s\x1b[0m&quot;, `异常抛出${err.toString()}`);
        conn.end();
        reject(err);
      }
      stream.on(&#39;close&#39;, async (code, signal) =&gt; { // 结束 code: 代码 signal: 信号
        if (code !== 0) {
          console.log(&quot;\x1B[35m%s\x1b[0m&quot;, `脚本异常退出code: ${code}，异常信号signal:${signal}`)
          conn.end();
          reject(&quot;\x1B[35m%s\x1b[0m&quot;, `脚本异常退出code: ${code}，异常信号signal:${signal}`);
        }
        // 程序执行成功
        // 自己的业务逻辑....记得conn.end();
        resolve();
      }).on(&#39;data&#39;, async data =&gt; { // 数据 程序执行中echo出的数据
        // console.log(&quot;\x1B[35m%s\x1b[0m&quot;,`echo出的数据${data.toString()}`);
      }).stderr.on(&#39;data&#39;, async data =&gt; { // 标准错误
        console.log(&quot;\x1B[35m%s\x1b[0m&quot;, `标准错误${data.toString()}`);
        conn.end();
        reject(data);
      });
    });
  })
}

function deleteFolder(path) {
  let files = [];
  if (fs.existsSync(path)) {
    files = fs.readdirSync(path);
    files.forEach(function (file, index) {
      let dirPath = path + &quot;/&quot; + file;
      if (fs.statSync(dirPath).isDirectory()) {
        deleteFolder(dirPath);
      } else {
        fs.unlinkSync(dirPath);
      }
    });
    fs.rmdirSync(path);
  }
