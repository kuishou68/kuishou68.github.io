12:T998,<p><strong>原因:</strong> 因为 navigator.clipboard
只能在安全网络环境中才能使用，也就是，安全域包括本地访问与开启TLS安全认证的地址，如 <code>https</code> 协议的地址、<code>127.0.0.1</code> 或 <code>localhos</code>才能正常使用，那在非安全环境，比如<code>HTTP</code>
协议的地址，用服务器IP访问的模拟环境又该如何使用呢？</p>
<p><strong>解决方案</strong>：</p>
<pre><code>&lt;template&gt;
    &lt;div&gt;
        &lt;span&gt;{{ text }}&lt;/span&gt;
        &lt;el-button type=&quot;text&quot; @click=&quot;handleCopy(text)&quot;&gt;复制&lt;/el-button&gt;
    &lt;/div
&lt;/template&gt;
&lt;script&gt;
export default {
  data() {
      text: &quot;文本文本文本文本&quot;
  },
  methods: {
    handleCopy(content) {
      // navigator clipboard 需要https等安全上下文
      if (navigator.clipboard &amp;&amp; window.isSecureContext) {
        // navigator clipboard 向剪贴板写文本
        navigator.clipboard.writeText(content);
        this.$message({
          message: &quot;复制成功&quot;,
          type: &quot;success&quot;
        });
      } else {
        // 创建text area
        let textArea = document.createElement(&quot;textarea&quot;);
        textArea.value = content;
        // 使text area不在viewport，同时设置不可见
        textArea.style.position = &quot;absolute&quot;;
        textArea.style.opacity = 0;
        textArea.style.left = &quot;-999999px&quot;;
        textArea.style.top = &quot;-999999px&quot;;
        document.body.appendChild(textArea);
        textArea.focus();
        textArea.select();
        return new Promise((res, rej) =&gt; {
          // 执行复制命令并移除文本框
          document.execCommand(&#39;copy&#39;) ? res() : rej();
          textArea.remove();
          this.$message({
            message: &quot;复制成功&quot;,
            type: &quot;success&quot;
          });
        });
      }
    }
  }
}
&lt;/script&gt;</code></pre>
