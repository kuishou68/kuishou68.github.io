int countPrimes(int n) {
    // дј е…Ҙзҡ„йқһиҙҹж•ҙж•°пјҢеӯҳж”ҫеҲ°ж•°з»„дёӯ
    boolean[] isPrime = new boolean[n];
    // е…ЁйғЁдј е…ҘдёәTrue
    Arrays.fill(isPrime, true);
    for (int i = 2; i * i < n; i++) 
        // еҰӮжһңдёәж•°з»„,еҲ йҷӨжүҖжңүж•°з»„зҡ„еҖҚж•°
        if (isPrime[i]) 
            // еҺ»йҷӨж•°з»„зҡ„еҖҚж•°
            for (int j = i * i; j < n; j += i) 
				// еҲ йҷӨж“ҚдҪң
                isPrime[j] = false;
    // з»ҹи®ЎжңүеҮ дёӘtrue
    int count = 0;
    // йҒҚеҺҶtrueзҡ„дёӘж•°
    for (int i = 2; i < n; i++)
        // еҰӮжһңдёәtrueпјҢcount+1
        if (isPrime[i]) count++;
    
    return count;
}

        // еҰӮжһңдёҚжҳҜ еӯҳж”ҫйҒҚеҺҶз»“жһңзҡ„ж•°з»„signsдёӯ еүҚдёҖдҪҚ
        if (!signs[i - 1]) {
            // з»ҹи®ЎеҲ°иҙЁж•°пјҢж•°йҮҸ+1
            count++;
			// з»§з»ӯйҒҚеҺҶжҹҘжүҫiеҗҺйқў iзҡ„еҖҚж•°[4,6,8,10...],

                // жүҫеҲ°еҗҺж Үи®°дёәtrue
                signs[j - 1] = true;
            }
        }
    }
    return count;
};

var countPrimes = function(n) {
    // жҺҘж”¶йқһиҙҹж•ҙж•°,з”Ё1жқҘеЎ«е……ж•°з»„дёӯиө·е§ӢдҪҚзҪ®
    const signs = new Array(n).fill(1);
    const primes = [];
    for(let i = 2; i < n; ++i){
        if(signs[i]){
			primes.push(i);
        }
        // йҒҚеҺҶ jзҡ„еҖје°ҸдәҺprimesй•ҝеәҰе’Ңж•°з»„дёӯзҡ„еҖје°ҸдәҺn 
        for(let j = 0; j < primes.length && i * primes[j] < n; ++j){
            signs[i* primes[j]] = 0;
			if(i % primes[j] === 0){
				break;
            }
        }
    }
    return primes.length;
};
